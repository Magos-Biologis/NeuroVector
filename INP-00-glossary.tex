%%%%%%%%%% Commands

\gls[<options>]{<label>}[<insert>] % works for both acronym or glossery term
\glspl[<options>]{<label>}[<insert>] % creates a plural form of term

\Gls[<options>]{<label>}[<insert>] % capitalizes the first character of the thing
\Glspl[<options>]{<label>}[<insert>]  % same thing, but plural

\glssymbol[<options>]{<label>}[<insert>]  % summons the symbol in the case a symbol is assigned

\end{comment}
